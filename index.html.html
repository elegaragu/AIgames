        // --- Game state variables ---

        let currentDoorIndex = 0;

        const currentDoorElement = document.getElementById('current-door');

        const questionTextElement = document.getElementById('question-text');

        const answersContainer = document.getElementById('answers-container');

        const messageElement = document.getElementById('message');

        const doorGrid = document.getElementById('door-grid');

        // Function to initialize the door grid

        function initializeDoorGrid() {

            try {

                doorGrid.innerHTML = '';

                for (let i = 0; i < questions.length; i++) {

                    const door = document.createElement('div');

                    door.classList.add('door', 'relative');

                    door.id = `door-grid-${i}`;

                    door.textContent = i + 1;

                    doorGrid.appendChild(door);

                }

            } catch (error) {

                console.error("Error initializing door grid:", error);

            }

        }

        // Function to load the current door's question

        function loadQuestion() {

            try {

                if (currentDoorIndex < questions.length) {

                    const currentQuestion = questions[currentDoorIndex];

                    questionTextElement.textContent = currentQuestion.question;

                    answersContainer.innerHTML = '';

                    messageElement.textContent = '';

                    // Create buttons for each answer

                    currentQuestion.answers.forEach(answer => {

                        const button = document.createElement('button');

                        button.classList.add(

                            'answer-button',

                            'w-full',

                            'py-3',

                            'px-6',

                            'rounded-full',

                            'border-2',

                            'border-gray-300',

                            'text-gray-700',

                            'font-semibold',

                            'bg-gray-200'

                        );

                        button.textContent = answer;

                        button.onclick = () => checkAnswer(answer);

                        answersContainer.appendChild(button);

                    });

                   

                    // Update the displayed door number

                    currentDoorElement.textContent = currentDoorIndex + 1;

                } else {

                    // Game completed

                    endGame();

                }

            } catch (error) {

                console.error("Error loading question:", error);

            }

        }

        // Function to check the user's answer

        function checkAnswer(selectedAnswer) {

            try {

                const currentQuestion = questions[currentDoorIndex];

                const allButtons = answersContainer.querySelectorAll('button');

                // Disable all buttons to prevent multiple clicks

                allButtons.forEach(button => button.disabled = true);

                if (selectedAnswer === currentQuestion.correct) {

                    // Correct answer logic

                    messageElement.textContent = "¡Respuesta correcta! ¡Puerta desbloqueada!";

                    messageElement.classList.add('text-green-600');

                   

                    // Find the button for the correct answer and style it

                    allButtons.forEach(button => {

                        if (button.textContent === selectedAnswer) {

                            button.classList.remove('bg-gray-200', 'border-gray-300', 'text-gray-700');

                            button.classList.add('correct', 'bg-green-500', 'border-green-600', 'text-white');

                        }

                    });

                    // Animate the main door and the one in the grid

                    const mainDoor = document.getElementById('current-door');

                    const gridDoor = document.getElementById(`door-grid-${currentDoorIndex}`);

                   

                    if (mainDoor) {

                        mainDoor.classList.add('unlocked');

                        const unlockedMessage = document.createElement('div');

                        unlockedMessage.classList.add('door-unlocked-message');

                        unlockedMessage.textContent = "¡Abierta!";

                        mainDoor.appendChild(unlockedMessage);

                    }

                    if (gridDoor) {

                        gridDoor.classList.add('unlocked');

                    }

                   

                    // Advance to the next door after a delay

                    setTimeout(() => {

                        currentDoorIndex++;

                        if (currentDoorIndex < questions.length) {

                            // Reset styles for the next door

                            document.getElementById('current-door').classList.remove('unlocked');

                            loadQuestion();

                        } else {

                            endGame();

                        }

                    }, 2000);

                } else {

                    // Incorrect answer logic

                    messageElement.textContent = "Respuesta incorrecta. Inténtalo de nuevo.";

                    messageElement.classList.add('text-red-600');

                   

                    // Style the incorrect button and highlight the correct one

                    allButtons.forEach(button => {

                        if (button.textContent === selectedAnswer) {

                            button.classList.remove('bg-gray-200', 'border-gray-300', 'text-gray-700');

                            button.classList.add('incorrect', 'bg-red-500', 'border-red-600', 'text-white');

                        }

                        if (button.textContent === currentQuestion.correct) {

                            button.classList.remove('bg-gray-200', 'border-gray-300', 'text-gray-700');

                            button.classList.add('correct', 'bg-green-500', 'border-green-600', 'text-white');

                        }

                    });

                   

                    // Re-enable buttons after a short delay

                    setTimeout(() => {

                         allButtons.forEach(button => {

                            button.disabled = false;

                            button.classList.remove('correct', 'incorrect', 'bg-green-500', 'bg-red-500', 'border-green-600', 'border-red-600', 'text-white');

                            button.classList.add('bg-gray-200', 'border-gray-300', 'text-gray-700');

                        });

                        messageElement.textContent = '';

                    }, 2000);

                }

            } catch (error) {

                console.error("Error checking answer:", error);

            }

        }

        // Function to handle the end of the game

        function endGame() {

            try {

                questionTextElement.textContent = "¡Felicidades! ¡Has desbloqueado las 10 puertas y completado el juego!";

                answersContainer.innerHTML = '';

                messageElement.textContent = '';

                currentDoorElement.style.display = 'none';

                doorGrid.style.display = 'none';

                // Add a restart button

                const restartButton = document.createElement('button');

                restartButton.classList.add(

                    'w-full', 'py-4', 'px-6', 'rounded-full', 'border-2', 'border-blue-500',

                    'bg-blue-600', 'text-white', 'font-bold', 'text-lg', 'mt-8', 'answer-button'

                );

                restartButton.textContent = "Jugar de Nuevo";

                restartButton.onclick = () => {

                    currentDoorIndex = 0;

                    currentDoorElement.style.display = 'flex';

                    doorGrid.style.display = 'grid';

                    initializeDoorGrid();

                    loadQuestion();

                };

                answersContainer.appendChild(restartButton);

            } catch (error) {

                console.error("Error ending game:", error);

            }

        }

        // Initialize the game when the window loads

        window.onload = function() {

            initializeDoorGrid();
